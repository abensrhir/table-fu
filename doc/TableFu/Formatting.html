def last_name(name)
  name.strip!
  if name.match(/\s(\w+)$/)
    $1
  else
    name
  end
end

def last_name_first_name(name)
  last = last_name(name)
  first = name.gsub(last, '').strip    
  "#{last}, #{first}"
end
